<programa> ::= <sentencias>

<sentencias> ::= <sentencia> 
               | <sentencia> <sentencias>

<sentencia> ::= <declaracion_matriz>
              | <operacion_producto>
              | <imprimir>

<declaracion_matriz> ::= MATRIX ID '=' <matriz>

<matriz> ::= '[' <filas> ']'
<filas> ::= <fila> 
          | <fila> ',' <filas>

<fila> ::= '[' <numeros> ']'
<numeros> ::= NUM 
            | NUM ',' <numeros>

<operacion_producto> ::= ID '=' ID '*' ID

<imprimir> ::= PRINT ID
